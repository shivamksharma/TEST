// Send an OTP with policy-aware routing
import { VoizagClient } from '@voizag/sdk';

const client = new VoizagClient({
  apiKey: process.env.VOIZAG_KEY,
  region: 'us-east',
});

await client.messages.send({
  to: '+1-555-0101',
  channel: 'sms',
  template: 'otp_login',
  metadata: {
    policyTag: 'banking_otp',
    sessionId: 'a4b1c2',
  },
});
